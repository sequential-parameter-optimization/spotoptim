# requirements-docs.txt
# =============================================================================
# Documentation build dependencies for spotoptim (Quarto + quartodoc)
#
# IMPORTANT: Use Python 3.13 to run these tools (NOT 3.14).
# quartodoc 0.11.x uses pydantic v1 which is incompatible with Python 3.14+.
#
# Local usage:
#   python3.13 -m venv .venv-docs
#   source .venv-docs/bin/activate
#   pip install -r requirements-docs.txt
#
# PURPOSE: This file installs ONLY the tools needed to BUILD and RENDER docs.
#          Runtime dependencies (numpy, torch, scipy, etc.) are declared in
#          pyproject.toml. When `execute.freeze: auto` is active in _quarto.yml
#          the GitHub Action does not execute code cells, so the full runtime
#          stack is NOT required here.
#
# UPDATING: Run `pip install --upgrade <pkg>` locally, test the build, then
#           update the pinned version here.  Use `pip-compile requirements-docs.in`
#           with pip-tools for a fully-reproducible lock.
# =============================================================================

# ── Quarto Python integration ─────────────────────────────────────────────────
# Required for Quarto to execute .qmd cells via the Jupyter kernel.
jupyter==1.1.1

# ── API documentation generator ───────────────────────────────────────────────
# quartodoc reads Google-style docstrings via griffe and emits .qmd stubs.
# https://machow.github.io/quartodoc/
quartodoc==0.11.1

# ── Docstring parser (griffe is a quartodoc dependency, pinned explicitly) ────
# griffe 2.0 broke quartodoc 0.11.x's md_renderer (DocstringSectionFunctions).
# Pin to <2.0 until quartodoc ships a compatible release.
griffe<2.0

# ── Minimal stubs needed for quartodoc to import spotoptim without error ──────
# quartodoc imports the package to resolve type annotations; keep these in
# sync with the lower bounds in pyproject.toml.
numpy>=1.24.3
scipy>=1.10.1
scikit-learn>=1.3.0
